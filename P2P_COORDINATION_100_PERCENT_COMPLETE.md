```bash

# Multi-tower federation
biomeos deploy templates/multi-tower-federation.biome.yaml

# Full ecosystem

```

### Use in Code

```rust
use biomeos_core::p2p_coordination::{
    BeardogSecurityAdapter,
    SongbirdDiscoveryAdapter,
    P2PCoordinator,
};

// Discover primals by capability
let security = discover_by_capability("security").await?;
let discovery = discover_by_capability("discovery").await?;

// Create coordinator
let coordinator = P2PCoordinator::new(security, discovery);

// Coordinate!
let tunnel = coordinator.create_btsp_tunnel(...).await?;
```
